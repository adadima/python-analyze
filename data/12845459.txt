n = int(input())
s = list(map(int, input().split()))
cnt = 0
while True:
    b = s[0]
    s = s[1:]
    a = max(s)
    s.reverse()
    s.append(b)
    s.reverse()
    b = s.index(a)
    if s[0] < a:
        cnt += 1
        s[0] += 1
        s[b] -= 1
    elif s[0] == a:
        cnt += 1
        break
    else:
        break
print(cnt)
