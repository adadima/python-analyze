int solve(int ll, int rr, int n)
{
 int count_ = 0;
 if (ll + 1 <= n)
 {
  count_++;
  count_ += solve(ll + 1, rr, n);
 }
 if (rr + 1 <= n)
 {
  count_++;
  count_ += solve(ll, rr + 1, n);
 }
 return count_;
}

int count_1 = 0, count_ = 0, ll = 0, rr = 0;



int fact(int n)
{
 int res = 1;
 int i;
 for (i = 1; i <= n; i++)
 {
  res *= i;
 }
 return res;
}
