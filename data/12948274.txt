n, m = [int(i) for i in input().split()]
a = [int(i) % m for i in input().split()]
b = [[0] * m for i in range(n)]
b[0][0] = 2
b[0][a[0]] = 1
for i in range(1, n):
    for j in range(m):
        if b[i - 1][j] > 0:
            b[i][j] = b[i - 1][j]
            if j + a[i] >= m:
                b[i][j + a[i] - m] = 1
            else:
                b[i][j + a[i]] = 1
if b[n - 1][0] == 1:
    print('YES')
else:
    print('NO')
