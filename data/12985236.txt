import sys
import math

def factors(x):
    i = 1
    n = min(x/2, math.sqrt(x))
    while i <= n:
        if x % i == 0:
            yield i
        i = i + 1
    yield x

(n, x) = map(int, sys.stdin.readline().split())

if n*n < x:
    print(0)
    exit()

occurrences = [(row, x/row) for row in filter(lambda f: f <= n, factors(x))
                            if  x / row <= n]

print(len(occurrences))
