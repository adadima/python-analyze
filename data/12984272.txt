n = int(input())
arr = [int (i) for i in input().split(" ")]
lim = arr.pop(0)

ans = 0
for i in range(lim, 1000):
    sum = 0
    for j in arr:
        sum += max(0, j-i)
    if i - lim > sum :
        ans = i - lim
        break
        
print (ans)
