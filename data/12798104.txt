# find the index which indicates the maximum 
def find_max_index(_list):
    i = 0
    l = len(_list)
    maximum = 0
    max_index = 0
    while i < l:
        if maximum < _list[i]:
            maximum = _list[i]
            max_index = i
        i = i + 1
    return max_index
