l = list(map(int, input().split()))
a = l[0]
b = l[1]

c = 0

if a * a < b or b == 0:
    print(0)
else:
    for i in range(2, a // 2 + 1):
        if i * i == b:
            break
        elif b % i == 0 and b / i <= a:
            c += 2
    for i in range(1, a + 1):
        if i * i == b:
            c += 1

    print(c)
