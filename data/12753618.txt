n = int(input())



k = [int(s) for s in input().split()]

l = k[0]
k.remove(l)

k.sort(reverse=1)
res =0
while l<=k[0]:
    res+=1
    l+=1
    k[0]-=1
    i= 0
    while k[i]<k[i+1]:
        g = k[i]
        k[i]=k[i+1]
        k[i+1]=g
        i+=1
        if i >= n-2:
            break

print(res)
