n, m = map(int, input().split())
a = list(map(lambda x: x%m, map(int, input().split())))
h = {}

for el in a:
    h[el] = h.get(el, 0) + 1

for v in h.values():
    if v % m == 0:
        print("YES")
        exit()

possible = set([])
for el in a:
    new = set([el])
    for candidate in possible:
       new.add((candidate + el)%m)

    possible |= new
    if 0 in possible:
        print("YES")
        exit()
print("NO")
