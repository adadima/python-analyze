import itertools

n,m = map(int, input().split(" "))
l = [[] for i in range(n)]
for i in range(m):
    a,b = sorted(list(map(int, input().split(" "))))
    l[a].append(b)
result = 0
for i in range(n):
    for x,y in itertools.combinations(l[i],2):
        if y-1 in l[x-1]:
            result += 1
if result == 0:
    result = -1
print(result)
