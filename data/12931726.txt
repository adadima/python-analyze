def isPrime(n):
    for i in range(2,int(n**0.5)+1):
        if n%i==0:
            return False

    return True
(n,m)=(int(i) for i in input().split())
k=0
for j in range(1,n+1):
    for z in range(j,m//j):
        if(j*z==m):
            if(j!=z):
                k=k+2
            else:
                k=k+1
print(k)
