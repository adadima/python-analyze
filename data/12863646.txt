mod = 10 ** 9 + 7

def go(n, l, r):
    global mod
    if l < 0 and r < 0:
        return 0
    if l == r == 0:
        return 1
    if l <= 0:
        return (1 + go(n, l, r - 1)) % mod
    if r <= 0:
            return (1 + go(n, l - 1, r)) % mod    
    if l == r:
        return (2 * go(n - 1, l - 1, r) + 1) % mod
    return (1 + go(n - 1, l - 1, r) + go(n - 1, l, r - 1)) % mod

n = int(input())
print(go(2 * n, n , n))
